字符函数
            
                int isalnum(int c);//检查字符是否为字母或数字
                int isalpha(int c);//检查字符是否为字母
                int isdigit(int c);//检查字符是否为数字
                int islower(int c);//检查一个字符是否是小写字母
                int ispunct(int c);//检查一个字符是否是标点字符
                int isspace(int c);//检查一个字符是否是空白字符(空格，回车，换行，制表符，进纸符)
                int isupper(int c);//检查一个字符是否是大写字母
                int tolower(int c);//将字符转换成小写
                int toupper(int c);//将字符转换成大写
数学函数
            
                //计算浮点数值 num 的绝对值
                float       abs( float num );
                double      abs( double num );
                long double abs( long double num );


                //计算 num 的最接近整数（以浮点数格式），中点情况舍入为远离零
                float       round ( float num );
                double      round ( double num );
                long double round ( long double num );
